class HighlightDemo<T : Number>(
    id: Int,
    state: T,
    val map: StringMap
) : AbstractProcessor<T>(id, state), Executable {

    companion object Factory {
        @JvmStatic
        fun createInstance(): HighlightDemo<Int> {
            return HighlightDemo(
                id = 42,
                state = 0,
                map = emptyMap()
            )
        }
    }

    init {
        require(id >= 0) { "Initialization block string literal" }
    }

    operator fun String?.plus(other: T): String {
        val safeString = this ?: "default_fallback"
        return "$safeString -> ${other.toLong()}"
    }

    @Throws(Exception::class)
    override suspend fun executeCoroutine(input: String?): ResultState {
        var mutableLocal = 10
        mutableLocal += 5

        val characterLiteral = 'K'

        val rawStringLiteral = """
            This is a raw string.
            Escape sequences like \n or \t are just text here.
        """.trimIndent()

        val standardString = "Escaped \t \n string \\ \u00A9"

        outerLoop@ for (index in 0..mutableLocal step 2) {
            if (index == 5) break@outerLoop

            when (index) {
                1 -> println(characterLiteral)
                else -> continue@outerLoop
            }
        }

        val closureResult = { x: Int, y: Int -> x + y }(1, 2)
        val methodReference = ::doWork

        try {
            val isInputValid = input?.isNotEmpty() == true
            if (!isInputValid) throw IllegalStateException("Invalid state")
        } catch (exception: IllegalStateException) {
            exception.printStackTrace()
        } finally {
            mutableLocal = 0
        }

        val unknownData: Any = "Test Data"
        if (unknownData is String) {
            println(unknownData.length)
        }

        return ResultState.Success("Process complete")
    }

    override fun doWork(vararg arguments: String) {
        arguments.forEach {
            println("Argument is $it")
        }
    }

    inline fun <R> typeSafeCast(
        crossinline block: () -> Any
    ): R? {
        val result = block()
        return result as? R
    }
}
